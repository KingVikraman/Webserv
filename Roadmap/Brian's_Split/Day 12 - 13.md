**Modify `HttpRequest.cpp`:**
```cpp
bool HttpRequest::parse(const std::string& raw_request) {
    std::istringstream stream(raw_request);
    std::string line;
    
    // Parse request line
    std::getline(stream, line);
    if (!_parseRequestLine(line))
        return false;
    
    // Parse headers
    while (std::getline(stream, line)) {
        if (line == "\r" || line.empty())
            break;
        
        if (!_parseHeader(line))
            return false;
    }
    
    // Parse body if Content-Length exists
    if (hasHeader("Content-Length")) {
        int content_length = std::atoi(getHeader("Content-Length").c_str());
        
        if (content_length > 0) {
            _body.reserve(content_length);
            char c;
            for (int i = 0; i < content_length && stream.get(c); i++) {
                _body += c;
            }
        }
    }
    
    return true;
}
```

#include "../src/HttpRequest.hpp"
#include <iostream>

int main() {
    HttpRequest req;
    
    std::string test = 
        "POST /upload HTTP/1.1\r\n"
        "Host: localhost\r\n"
        "Content-Type: text/plain\r\n"
        "Content-Length: 11\r\n"
        "\r\n"
        "Hello World";
    
    if (!req.parse(test)) {
        std::cout << "✗ Parsing failed\n";
        return 1;
    }
    
    if (req.getMethod() == "POST")
        std::cout << "✓ Method is POST\n";
    else
        std::cout << "✗ Method wrong: " << req.getMethod() << "\n";
    
    if (req.getBody() == "Hello World")
        std::cout << "✓ Body correct\n";
    else
        std::cout << "✗ Body wrong: '" << req.getBody() << "'\n";
    
    if (req.getBody().length() == 11)
        std::cout << "✓ Body length correct\n";
    else
        std::cout << "✗ Body length wrong: " << req.getBody().length() << "\n";
    
    return 0;
}
