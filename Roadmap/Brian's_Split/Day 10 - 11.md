bool HttpRequest::_parseHeader(const std::string& line) {
    size_t colon_pos = line.find(':');
    
    if (colon_pos == std::string::npos)
        return false;
    
    std::string key = line.substr(0, colon_pos);
    std::string value = line.substr(colon_pos + 1);
    
    // Trim whitespace from value
    size_t start = value.find_first_not_of(" \t");
    size_t end = value.find_last_not_of(" \t\r\n");
    
    if (start != std::string::npos && end != std::string::npos)
        value = value.substr(start, end - start + 1);
    
    _headers[key] = value;
    return true;
}

bool HttpRequest::parse(const std::string& raw_request) {
    std::istringstream stream(raw_request);
    std::string line;
    
    // Parse request line
    std::getline(stream, line);
    if (!_parseRequestLine(line))
        return false;
    
    // Parse headers
    while (std::getline(stream, line)) {
        // Blank line marks end of headers
        if (line == "\r" || line.empty())
            break;
        
        if (!_parseHeader(line))
            return false;
    }
    
    // Rest is body
    std::string body_content;
    while (std::getline(stream, line)) {
        body_content += line + "\n";
    }
    _body = body_content;
    
    return true;
}

#include "../src/HttpRequest.hpp"
#include <iostream>

int main() {
    HttpRequest req;
    
    std::string test = 
        "GET /index.html HTTP/1.1\r\n"
        "Host: localhost:8080\r\n"
        "User-Agent: Mozilla/5.0\r\n"
        "Accept: text/html\r\n"
        "Content-Type: application/json\r\n"
        "\r\n";
    
    if (!req.parse(test)) {
        std::cout << "✗ Parsing failed\n";
        return 1;
    }
    
    // Test header retrieval
    if (req.getHeader("Host") == "localhost:8080")
        std::cout << "✓ Host header correct\n";
    else
        std::cout << "✗ Host header wrong: " << req.getHeader("Host") << "\n";
    
    if (req.getHeader("User-Agent") == "Mozilla/5.0")
        std::cout << "✓ User-Agent header correct\n";
    else
        std::cout << "✗ User-Agent header wrong\n";
    
    if (req.getHeader("Accept") == "text/html")
        std::cout << "✓ Accept header correct\n";
    else
        std::cout << "✗ Accept header wrong\n";
    
    if (req.hasHeader("Content-Type"))
        std::cout << "✓ hasHeader works\n";
    else
        std::cout << "✗ hasHeader failed\n";
    
    if (req.getHeader("NonExistent").empty())
        std::cout << "✓ Non-existent header returns empty\n";
    else
        std::cout << "✗ Non-existent header should be empty\n";
    
    return 0;
}
